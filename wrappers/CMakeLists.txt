#
# Dependent on variables:
#     - LANG - target language.
#
# Define variables:
#     - WRAPPED_LIB_NAME - name of the wrapped library.
#

cmake_minimum_required (VERSION 3.2 FATAL_ERROR)

project (wrappers)

# Define variables
set (WRAPPED_LIB_NAME virgil_crypto)

# Add include directory of wrapped module
include_directories (${${WRAPPED_LIB_NAME}_SOURCE_DIR}/include)
include_directories (${${WRAPPED_LIB_NAME}_BINARY_DIR}/include)
include_directories (swig)

# Add wrappers
set (CMAKE_INSTALL_RPATH "@loader_path")
set (LANG_DIR ${LANG})
if (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${LANG_DIR}")
    add_subdirectory (${LANG_DIR})
else ()
    message (STATUS "No special wrappers for the language: " ${LANG})
endif ()
